            console.log(1);
            console.log(2);
        }
        function secondSyncLog (callbackFunction) {
            setTimeout(function() {
                console.log(3);
                callbackFunction();
            }, 2000)
        }
        function lastSyncLog () {
            console.log(4);
            console.log(5);
        }
        firstSyncLog();
        secondSyncLog(lastSyncLog);

pure function:
1- Given the same input, it should always return the same output.
2- The function should not cause any side effects outside of its scope. run and check logs 

        function pureFunction (num1, num2) {
            let tenPlus = 10;
            return Math.pow(num1, num2) + tenPlus;
            console.log(pureFunction());
        }
